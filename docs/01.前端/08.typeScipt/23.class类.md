### class类的基本用法（继承和类型约束implements）
```ts
interface Options {
    el: string | HTMLElement
}
interface VueCls {
    options: Options
    init(): void
}

class Vue implements VueCls{
    options: Options
    init(): void
}
new Vue({
    el: "#app"
})
```
